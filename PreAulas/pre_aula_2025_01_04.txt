select * from movies;
select * from directors;

-- Mostra informações relacionadas entre duas tabelas (boa prática)
-- Ele vai comparar a chavbe primária da tabela diretor com a chave estrangeira da tabela filmes;
-- O Where vai especificar qual diretor fez quais filmes (no caso o diretor com id = 1);

Select *
from movies as m
	inner join directors as d
	on m.director_id = d.id
where m.director_id = 1;

-- As duas primeiras linhas do comando acaba em um Cross Join;
-- A condição "força" uma situação em que não mostra as repetições (não é uma boa prática);

select *
from movies as m, directors as d
where m.director_id = d.id;

-- Este comando faz uma nova condição para que faça, aparentemente, a mesma coisa
-- que o primeiro comando mostrado, mas, como no segundo, ele "força" uma condição
-- para que seja semelhante ao comando usado como boa prática

select *
from movies as m, directors as d
where m.director_id = d.id
	and m.director_id = 1;

-- Em resumo, o primeiro comando de junção (Inner Join) é uma boa prática que, não vai gerar um Cross Join
-- Nos outros comandos, é uma versão mais implícita, o qual não é uma boa prática, do primeiro comando
